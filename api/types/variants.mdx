title: Variants
description: Extract variant types from composed styles
---

The `Variants` type helper extracts the variant types from a composed style, allowing you to type component props that accept variants.

## Import

```typescript
import type { Variants } from '@tokenami/css';
```

## Signature

```typescript
type Variants<T> = T extends TokenamiComposeOutput<infer V> 
  ? CSSVariants<V> 
  : {}
```

<ParamField path="T" type="TokenamiComposeOutput">
  A composed style created with `css.compose()`.
</ParamField>

<ResponseField name="Variants<T>" type="object">
  Object type where keys are variant names and values are the possible variant values.
</ResponseField>

## Usage

### Basic Variant Extraction

Extract variant types from a composed style:

```typescript
import { css } from '@tokenami/css';
import type { Variants } from '@tokenami/css';

const button = css.compose({
  '--padding': 'var(--space_3)',
  
  variants: {
    variant: {
      primary: { '--background-color': 'var(--color_primary)' },
      secondary: { '--background-color': 'var(--color_secondary)' }
    },
    size: {
      sm: { '--padding': 'var(--space_2)' },
      md: { '--padding': 'var(--space_3)' },
      lg: { '--padding': 'var(--space_4)' }
    }
  }
});

type ButtonVariants = Variants<typeof button>;
// Result:
// {
//   variant?: 'primary' | 'secondary';
//   size?: 'sm' | 'md' | 'lg';
// }
```

### Component Props with Variants

Use extracted variants in component props:

```typescript
import { css } from '@tokenami/css';
import type { Variants } from '@tokenami/css';

const button = css.compose({
  '--padding': 'var(--space_3)',
  '--border-radius': 'var(--radius_md)',
  
  variants: {
    variant: {
      primary: {
        '--background-color': 'var(--color_primary)',
        '--color': 'var(--color_white)'
      },
      secondary: {
        '--background-color': 'var(--color_gray-200)',
        '--color': 'var(--color_gray-900)'
      },
      ghost: {
        '--background-color': 'transparent',
        '--color': 'var(--color_primary)'
      }
    },
    size: {
      sm: { '--padding': 'var(--space_2)' },
      md: { '--padding': 'var(--space_3)' },
      lg: { '--padding': 'var(--space_4)' }
    }
  }
});

interface ButtonProps extends Variants<typeof button> {
  children: React.ReactNode;
  onClick?: () => void;
}

function Button({ 
  children, 
  onClick,
  variant = 'primary',
  size = 'md'
}: ButtonProps) {
  const [cn, style] = button({ variant, size });
  
  return (
    <button className={cn()} style={style()} onClick={onClick}>
      {children}
    </button>
  );
}

// TypeScript enforces correct variant values
<Button variant="primary" size="lg">Click me</Button>  // ✅
<Button variant="danger" size="xl">Click me</Button>   // ❌ Type error
```

### With TokenamiStyle

Combine with `TokenamiStyle` for components that accept both variants and style overrides:

```typescript
import { css } from '@tokenami/css';
import type { Variants, TokenamiStyle } from '@tokenami/css';

const button = css.compose({
  '--padding': 'var(--space_3)',
  
  variants: {
    variant: {
      primary: { '--background-color': 'var(--color_primary)' },
      secondary: { '--background-color': 'var(--color_secondary)' }
    }
  }
});

interface ButtonProps extends Variants<typeof button> {
  children: React.ReactNode;
}

function Button({ 
  children, 
  variant = 'primary',
  style 
}: TokenamiStyle<ButtonProps>) {
  const [cn, buttonStyle] = button({ variant });
  
  return (
    <button className={cn()} style={buttonStyle(style)}>
      {children}
    </button>
  );
}

// Accepts both variants and style overrides
<Button 
  variant="primary" 
  style={{ '--font-size': 'var(--font-size_lg)' }}
>
  Click me
</Button>
```

### Boolean Variants

Boolean string variants are typed as actual booleans:

```typescript
const button = css.compose({
  variants: {
    disabled: {
      true: { '--opacity': '0.5' },
      false: { '--opacity': '1' }
    },
    loading: {
      true: { '--cursor': 'wait' },
      false: { '--cursor': 'pointer' }
    }
  }
});

type ButtonVariants = Variants<typeof button>;
// Result:
// {
//   disabled?: boolean;  // Not 'true' | 'false'
//   loading?: boolean;
// }

interface ButtonProps extends Variants<typeof button> {
  children: React.ReactNode;
}

function Button({ disabled, loading, children }: ButtonProps) {
  const [cn, style] = button({ disabled, loading });
  return <button className={cn()} style={style()}>{children}</button>;
}

// Use as booleans
<Button disabled loading>Click me</Button>
```

### Numeric Variants

Numeric string variants are typed as actual numbers:

```typescript
const icon = css.compose({
  '--size': 4,
  
  variants: {
    size: {
      16: { '--size': 2 },
      24: { '--size': 3 },
      32: { '--size': 4 },
      48: { '--size': 6 }
    }
  }
});

type IconVariants = Variants<typeof icon>;
// Result:
// {
//   size?: 16 | 24 | 32 | 48;  // Numbers, not strings
// }

interface IconProps extends Variants<typeof icon> {
  name: string;
}

function Icon({ name, size = 24 }: IconProps) {
  const [cn, style] = icon({ size });
  return <svg className={cn()} style={style()}>{/* ... */}</svg>;
}

// Use as numbers
<Icon name="check" size={32} />
```

### Optional Variants

All variants are optional by default:

```typescript
const button = css.compose({
  variants: {
    variant: {
      primary: { /* ... */ },
      secondary: { /* ... */ }
    }
  }
});

type ButtonVariants = Variants<typeof button>;
// Result: { variant?: 'primary' | 'secondary' }

// All valid:
button();
button({ variant: 'primary' });
button({ variant: 'secondary' });
```

### Reusing Variant Types

Extract and reuse variant types across multiple components:

```typescript
const button = css.compose({
  variants: {
    variant: {
      primary: { /* ... */ },
      secondary: { /* ... */ },
      danger: { /* ... */ }
    },
    size: {
      sm: { /* ... */ },
      md: { /* ... */ },
      lg: { /* ... */ }
    }
  }
});

type ButtonVariants = Variants<typeof button>;

// Use in multiple places
interface ButtonProps extends ButtonVariants {
  children: React.ReactNode;
}

interface ButtonGroupProps {
  variant?: ButtonVariants['variant'];
  size?: ButtonVariants['size'];
  buttons: Array<{ label: string; onClick: () => void }>;
}

function ButtonGroup({ variant, size, buttons }: ButtonGroupProps) {
  return (
    <div>
      {buttons.map(({ label, onClick }) => (
        <Button key={label} variant={variant} size={size} onClick={onClick}>
          {label}
        </Button>
      ))}
    </div>
  );
}
```

## Type Details

The `Variants` type:

- Extracts variant names and values from composed styles
- Makes all variants optional (prefixed with `?`)
- Converts boolean string literals (`'true'` | `'false'`) to `boolean`
- Converts numeric string literals to `number` types
- Returns empty object `{}` for non-composed styles
- Provides full IntelliSense support
